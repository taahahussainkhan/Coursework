#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <stdlib.h>
#include <wait.h>
#include <pthread.h>


void* f1(void* ptr)
{
	while(1)
		printf("Thread 1");

}


void* f2(void* ptr)
{
	while(1)
		printf("Thread 2");

}


int main()
{
	pthread_t tid1,tid2;

	pthread_create(&tid1,NULL,&f1,NULL);
	pthread_create(&tid2,NULL,&f2,NULL);

	pthread_join(tid1,NULL);
	pthread_join(tid2,NULL);

return 0;
}

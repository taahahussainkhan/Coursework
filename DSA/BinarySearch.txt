#include <iostream>


int main()
{
	const int N = 16;
	int A[N] = { 10, 11, 20, 30, 40, 50, 60, 70, 80, 90, 100, 110, 120, 130, 140, 150 };
	int low = 0, high = N - 1, mid = 0;
	int Value = 70;
	while (low <= high)
	{
		mid = (low + high) / 2;
		if (A[mid] == Value)
		{
			std::cout << mid << std::endl;
			break;
		}
		else if (A[mid] < Value)
			low = mid + 1;
		else
			high = mid - 1;
	}

	return 0;
}
